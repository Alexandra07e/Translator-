1. stm

TEST 1 - the if condition must be a scalar value
struct A{ 
    int x; 
};

int main(){
    struct A a;
    if (a){
        return 1;
    }
    return 0;
}


TEST 2 - the while condition must be a scalar value
struct B{
    int x;
};

int  main(){
    struct B b;
    while(b){
        return 1;
    }
    return 0;
}


TEST 3 - the return value must be a scalar value
int main(){
    int v[10];
    return v;
}


TEST 4 - a void function cannot return a value
void f(){
    return 1;
}


TEST 5 - a non-void function must return a value
int f(){
    return;
}


TEST 6 - cannot convert the return expression type to the function return type
struct A{
    int x;
};

struct A f(){
    return 1; //ar trebui sa returneze ceva de tip struct A , NU INT
}



2. exprAssign

TEST 7 - the assign destination must be a left-value
int main(){
    int x;
    3 = x;
    return 0;
}


TEST 8 - the assign destination cannot be constant
int v[3];

void f(int a[]){
    v = a;
}


TEST 9 - the assign destination must be scalar
struct A{
    int x;
};

struct A a;

void f(){    //Aici folosim o structură ca destinație. Este left-value, nu e constantă, dar nu este scalară.
    a = 1;
}


TEST 10 - the assign source must be scalar
int x;
int v[3];

void f(){  //Aici destinația e bună, x este scalar, dar sursa este vectorul v.
    x = v;
}

9. exprCast

TEST 18 - cannot convert to a struct type
struct A{ 
    int x; 
};

int main(){
    int x;
    x = (struct A)1;
    return 0;
}


TEST 19 - cannot convert a struct
struct A{
    int x;
};

void f(){
    struct A a;
    int x;
    x = (int)a;
}


TEST 20 - an array can be converted only to another array
void f(){
    int v[3];
    int x;
    x = (int)v;
}


TEST 21 - a scalar can be converted only to another scalar
void f(){
    int x;
    x = (int[3])x;
}

}



11. exprPostfix

TEST 23 - only an array can be indexed
int main(){
    int x;
    x[0] = 1;
    return 0;
}


TEST 24 - the index is not convertible to int
struct A{
    int x;
};

void f(){
    int v[3];
    struct A a;
    v[a] = 1;
}


TEST 25 - a field can only be selected from a struct
void f(){
    int x;
    int y;
    y = x.a;
}


TEST 26 - the structure %s does not have a field %s
struct A {
    int x;
};

void f() {
    struct A a;
    int z;
    z = a.y;
}



12. exprPrimary

TEST 27 - "undefined id: %s
int main(){
    x = 3;
    return 0;
}


TEST 28 - only a function can be called

    int x;
    x();
}


TEST 29 - too many arguments in function call

    return;
}

void f(){
    g(1, 2);
}


TEST 30 - in call, cannot convert the argument type to the parameter type
struct A{
    int x;
};

void g(struct A a){
    return;
}

void f(){
    g(1);
}


TEST 31 - too few arguments in function call

    return;
}

void f(){
    g(1);
}


TEST 32 - a function can only be called

    return;
}

void f(){
    g;
    return;
}
